def expired
    @bottle = Bottle.find(Bottle.get_bottle_id(bottle_id))
    if @bottle.expiration_date < DateTime.now
        return true
    else
        return false
    end
end

def verify
    @bottle = Bottle.find(Bottle.get_bottle_id(bottle_id))
    @patient = Visit.where(account_number: patient_id).first.patient
    if @patient.id == @bottle.patient.id
        return true
    else
        return false
    end
end
